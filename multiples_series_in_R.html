library(lubridate)
modelo2 <- dd %>%
  model( arima = ARIMA(value),
         ets = ETS(value),
         snaive =SNAIVE(value))
f1 <- modelo2 %>%
  forecast(h = "1 year")
f1 %>%
  autoplot(dd, level = 95, alpha = 0.5) +
  ggthemes::theme_pander()
